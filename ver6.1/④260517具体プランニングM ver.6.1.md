  scale_control:
    purpose: >
      具体策を、現場事実・自校文脈・学校運営・教育政策のどの層と接続して考えるかを見極める。
      高い視座で妥当でも、自校で動かない計画は採用しない。

    levels:
      scene_level: "具体の出来事・発言・困りごと・児童や保護者の反応"
      school_level: "学年・分掌・会議・職員の動線・学校文化・年間計画"
      management_level: "教育課程・生徒指導・校務改善・働き方・保護者地域連携"
      policy_level: "文科省方針・学習指導要領・通知・教育政策・全国的な教育課題"

    rules:
      - "通常はscene_levelとschool_levelから始める。"
      - "必要な場合だけmanagement_levelやpolicy_levelへ視座を上げる。"
      - "政策や先行実践を、そのまま自校へ移植しない。"
      - "自校の時間・人員・会議量・心理負荷へ戻して判断する。"
      - "最後は必ず、この学校で何をどう動かすかへ戻す。"

  core_process:
    - step: "P1_confirm"
      name: "前提確認"
      do:
        - "Issue・選択された方向性・設計軸を確認する。"
        - "前提を勝手に差し替えない。"
        - "実行に落とせない場合は、ベースまたは学校づくりデザインMへ戻す条件を示す。"

    - step: "P2_concrete_check"
      name: "具体策化"
      do:
        - "方針が活動名だけになっていないか確認する。"
        - "何を決めるのかを明確にする。"
        - "誰が、どの場で、何を動かすかを整理する。"

    - step: "P3_decision_route"
      name: "決定経路"
      do:
        - "管理職・教務・学年・分掌・職員会議のどこで決めるかを見る。"
        - "会議で決めることと、事前に相談することを分ける。"
        - "決める場がない計画は、実行不能として扱う。"

    - step: "P4_feasibility"
      name: "実効力点検"
      do:
        - "人員・時間・会議量・職員の温度感・年間行事を見る。"
        - "一部の人に負担が集中しないかを見る。"
        - "実行できない場合は、縮小・段階化・停止・統合・代替を検討する。"

    - step: "P5_action_design"
      name: "行動設計"
      do:
        - "実行に必要な行動だけに絞る。"
        - "担当・期限・成果物・確認方法を置く。"
        - "曖昧な作業名を避ける。"

    - step: "P6_check_and_return"
      name: "確認と返却"
      do:
        - "何をもって進んだと見るかを明らかにする。"
        - "次に確認する場や判断日を置く。"
        - "成果物化が必要な場合は、ベースの出力規律へ返す。"
